29  /**
30  * Wrapper class and C++ representation of KnoRBA `grid` type. KnoRBA `grid`
31  * is a multi-dimensional array of records. Both KGrid and KRecord are
32  * optimized together for high performance computing. KGrid takes full
33  * advantage of range arithmatics objects in KFoundation.
34  *
35  * All values of a grid are squizzed into a continues portion of memory for
36  * better allocation and cache performance. No KRecord object is allocated
37  * internally. Allocating a KRecord for each cell would consume too much
38  * uncessary memory, and also cause fragmentation, decreasing cache
39  * performance. Therefore the design pattern chosen for this object is
40  * *sliding record*, that is, only one KRecord is allocated and is slid
41  * over desired cells of the grid. However the KRecord is not allocated
42  * internally because it would make it impossible to read and write
43  * on the same grid using more than one thread. Therefore, eahc thread should
44  * allocate the slidding KRecord externally.
45  *
46  * KGrid itslef is an abstract class. Use one of the implementations provided
47  * in the same header:
48  *
49  * * KGridBasic -- internally allocated multi-dimensional array of cells.
50  * * KGridWindow -- a sub array of an externally allocated KGrid.
51  * * KGridVector -- a one dimensional dynamically resizable array of cells.
52  *
53  * The basic usage of KGrid with simple record type is as follows:
54  *
55  * #include <kfoundation/Tuple.h>
56  * #include <kfoundation/RangeIterator.h>
57  * using namespace kfoundation;
58  *
59  * Ptr<KGridType> gridType = new KGridType(KType::REAL, 2);
60  * Ptr<KGrid> grid = new KGridBasic(gridType, Tuple(100, 100));
61  * Ptr<KRecord> r = new KRecord(grid);
62  * for(RangeIterator i(grid->getSize()); i.hasMore(); i.next()) {
63  * grid->at(i, r)->setInteger(i.at(0));
64  * }
65  *
66  * The at() method slides the given KRecord on to the cell at the given index.
67  *
68  * grid->at(Tuple2D(11, 12), r);
69  * r->setInteger(124);
70  *
71  * In case the cell record type has more than one field, the name or index
72  * of the field can be supplied to KRecord::setXXX() and KRecord::getXX()
73  * methods.
74  *
75  * r->setInteger("year", 1981);
76  *
77  * @headerfile KGrid.h <knorba/type/KGrid.h>
78  */
